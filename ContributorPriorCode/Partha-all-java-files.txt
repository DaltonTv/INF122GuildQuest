import com.guildquest.model.*;

public class GuildQuestManager {

    //************* Important Assignment 3 changes ************//
    //***** CHANGED - Singleton: private static instance variable ****//
    private static GuildQuestManager instance;

    private List<User> users;
    private List<Realm> realms;
    private List<CampaignShare> shares;
    private WorldClock worldClock;
    private DayView dayView;


    //************* Important Assignment 3 changes ************//
    // ******** CHANGED - Singleton: constructor made private ****//
    private GuildQuestManager() {
        this.users = new ArrayList<>();
        this.realms = new ArrayList<>();
        this.shares = new ArrayList<>();
        this.worldClock = WorldClock.getInstance();
        this.dayView = new DayView();
    }

    //************* Important Assignment 3 changes ************//
    // ******* CHANGED - Singleton: global access point *******//
    public static GuildQuestManager getInstance() {
        if (instance == null) {
            instance = new GuildQuestManager();
        }
        return instance;
    }

    // End Factory Method.

    public User createUser(String userId, String name) {
        User user = new User(userId, name);
        users.add(user);
        return user;
    }

    //************* Important Assignment 3 changes ************//
    // CHANGED - Factory Method: use factory instead of direct instantiation
    private RealmFactory realmFactory = new StandardRealmFactory();

    public Realm createRealm(String realmId, String name,
                             String description, int offsetMinutes) {
        Realm realm = realmFactory.createRealm(realmId, name, description, offsetMinutes);
        realms.add(realm);
        return realm;
    }
    // end Factory Method


    public void shareCampaign(Campaign campaign, User sharedWith, PermissionLevel permission) {
        CampaignShare share = new CampaignShare(sharedWith, campaign, permission);
        shares.add(share);
    }

    public List<Campaign> getSharedCampaigns(User user) {
        List<Campaign> sharedCampaigns = new ArrayList<>();
        for (CampaignShare share : shares) {
            if (share.getSharedWith().equals(user)) {
                sharedCampaigns.add(share.getCampaign());
            }
        }
        return sharedCampaigns;
    }

    public List<QuestEvent> getDayEvents(Campaign campaign, WorldTime startTime) {
        return dayView.getEvents(campaign, startTime);
    }

    public List<User> getUsers() {
        return users;
    }

    public List<Realm> getRealms() {
        return realms;
    }

    public WorldClock getWorldClock() {
        return worldClock;
    }

    public User findUser(String userId) {
        for (User user : users) {
            if (user.getUserId().equals(userId)) {
                return user;
            }
        }
        return null;
    }

    public Realm findRealm(String realmId) {
        for (Realm realm : realms) {
            if (realm.getRealmId().equals(realmId)) {
                return realm;
            }
        }
        return null;
    }
}

package com.guildquest.model;

import java.util.ArrayList;
import java.util.List;

public class Campaign {
    private String campaignId;
    private String name;
    private Visibility visibility;
    private boolean archived;
    private User owner;
    private List<QuestEvent> events;

    // CHANGED - Observer: list of observers
    private List<Observer> observers = new ArrayList<>();

    public Campaign(String campaignId, String name, User owner) {
        this.campaignId = campaignId;
        this.name = name;
        this.owner = owner;
        this.visibility = Visibility.PRIVATE;
        this.archived = false;
        this.events = new ArrayList<>();
    }

    // CHANGED - Observer: add observer
    public void addObserver(Observer observer) {
        observers.add(observer);
    }

    // CHANGED - Observer: remove observer
    public void removeObserver(Observer observer) {
        observers.remove(observer);
    }

    // CHANGED - Observer: notify all observers
    private void notifyObservers(String eventTitle) {
        for (Observer o : observers) {
            o.update(this.name, eventTitle);
        }
    }

    // CHANGED - Observer: notify when event added
    public void addEvent(QuestEvent event) {
        events.add(event);
        notifyObservers(event.getTitle());
    }

    public void removeEvent(QuestEvent event) {
        events.remove(event);
    }

    public void rename(String newName) {
        this.name = newName;
    }

    public String getCampaignId() {
        return campaignId;
    }

    public String getName() {
        return name;
    }

    public Visibility getVisibility() {
        return visibility;
    }

    public void setVisibility(Visibility visibility) {
        this.visibility = visibility;
    }

    public boolean isArchived() {
        return archived;
    }

    public void setArchived(boolean archived) {
        this.archived = archived;
    }

    public User getOwner() {
        return owner;
    }

    public List<QuestEvent> getEvents() {
        return events;
    }
}

package com.guildquest.model;

public class CampaignShare {

    public CampaignShare(User sharedWith, Campaign campaign, PermissionLevel permission) {
        this.sharedWith = sharedWith;
        this.campaign = campaign;
        this.permission = permission;

    public Campaign getCampaign() {

    }

    public PermissionLevel getPermission() {
        return permission;
    }

    public void setPermission(PermissionLevel permission) {
        this.permission = permission;
    }

    public boolean canEdit() {
        return permission == PermissionLevel.COLLABORATIVE;

public class FixedOffsetRule implements LocalTimeRule {
    private int offsetMinutes;

    public FixedOffsetRule(int offsetMinutes) {
        this.offsetMinutes = offsetMinutes;
    }

    @Override
    public LocalTime toLocal(WorldTime worldTime) {
        int totalMinutes = worldTime.toTotalMinutes() + offsetMinutes;

        int days = totalMinutes / (24 * 60);
        int remainingMinutes = totalMinutes % (24 * 60);
        int hours = remainingMinutes / 60;
        int minutes = remainingMinutes % 60;

        if (hours < 0) {
            hours += 24;
            days -= 1;
        }
        if (minutes < 0) {
            minutes += 60;
            hours -= 1;
        }

        return new LocalTime(days, hours, minutes);
    }

    public int getOffsetMinutes() {
        return offsetMinutes;
    }
}

public class Inventory {
    private List<InventoryEntry> entries;

    public Inventory() {
        this.entries = new ArrayList<>();
    }

    public void addItem(Item item, int quantity) {
        for (InventoryEntry entry : entries) {
            if (entry.getItem().getItemId().equals(item.getItemId())) {
                entry.addQuantity(quantity);
                return;
            }
        }
        entries.add(new InventoryEntry(item, quantity));
    }

    public void removeItem(Item item, int quantity) {
        for (InventoryEntry entry : entries) {
            if (entry.getItem().getItemId().equals(item.getItemId())) {
                entry.removeQuantity(quantity);
                if (entry.getQuantity() <= 0) {
                    entries.remove(entry);
                }
                return;
            }
        }
    }

    public List<InventoryEntry> getEntries() {
        return entries;
    }

    public InventoryEntry findEntry(String itemId) {
        for (InventoryEntry entry : entries) {
            if (entry.getItem().getItemId().equals(itemId)) {
                return entry;
            }
        }
        return null;
    }
}

package com.guildquest.model;


public class WorldTime {
    private int days;
    private int hours;
    private int minutes;


    public WorldTime(int days, int hours, int minutes) {
        this.days = days;
        this.hours = hours;
        this.minutes = minutes;
        normalize(); // Ensure valid time (e.g., 25 hours becomes 1 day + 1 hour)
    }


    private void normalize() {
        // Convert excess minutes to hours
        if (minutes >= 60) {
            hours += minutes / 60;
            minutes = minutes % 60;
        }
        if (minutes < 0) {
            hours -= 1;
            minutes += 60;
        }

        // Convert excess hours to days
        if (hours >= 24) {
            days += hours / 24;
            hours = hours % 24;
        }
        if (hours < 0) {
            days -= 1;
            hours += 24;
        }
    }


    public WorldTime plusMinutes(int minutesToAdd) {
        return new WorldTime(this.days, this.hours, this.minutes + minutesToAdd);
    }


    public int compareTo(WorldTime other) {
        if (this.days != other.days) {
            return this.days - other.days;
        }
        if (this.hours != other.hours) {
            return this.hours - other.hours;
        }
        return this.minutes - other.minutes;
    }

    /**
     * Check if this time is before another time
     */
    public boolean isBefore(WorldTime other) {
        return compareTo(other) < 0;
    }

    /**
     * Check if this time is after another time
     */
    public boolean isAfter(WorldTime other) {
        return compareTo(other) > 0;
    }

    // Getters
    public int getDays() {
        return days;
    }

    public int getHours() {
        return hours;
    }

    public int getMinutes() {
        return minutes;
    }

    /**
     * Convert to total minutes (useful for calculations)
     */
    public int toTotalMinutes() {
        return (days * 24 * 60) + (hours * 60) + minutes;
    }

    @Override
    public String toString() {
        return String.format("Day %d, %02d:%02d", days, hours, minutes);
    }

    @Override
    public boolean equals(Object obj) {
        if (this == obj) return true;
        if (!(obj instanceof WorldTime)) return false;
        WorldTime other = (WorldTime) obj;
        return this.days == other.days &&
                this.hours == other.hours &&
                this.minutes == other.minutes;
    }
}

import java.util.List;

public class DayView implements TimelineView {

    @Override
    public List<QuestEvent> getEvents(Campaign campaign, WorldTime startTime) {
        List<QuestEvent> dayEvents = new ArrayList<>();

        WorldTime endOfDay = startTime.plusMinutes(24 * 60);

        for (QuestEvent event : campaign.getEvents()) {
            WorldTime eventStart = event.getStartTime();

            if (!eventStart.isBefore(startTime) && eventStart.isBefore(endOfDay)) {
                dayEvents.add(event);
            }
        }

        return dayEvents;
    }
}
